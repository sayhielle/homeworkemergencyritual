<script>

function summon(){

    let object1 = document.getElementById("object1").value;
    let object2 = document.getElementById("object2").value;
    let object3 = document.getElementById("object3").value;

    let result = document.getElementById("result");

    if(object1 == "an old candle that has seen better days." &&
       object2 == "your favorite pen (we all have one)" &&
       object3 == "the math book you're avoiding")
    {
        result.innerHTML =
        "A demon appears! It's a tiny one and kind of cute. It does your homework but the handwriting is really bad.";
    }

    else if(object1 == "a fresh cookie, still warm and all." ||
            object2 == "a fresh cookie, still warm and all." ||
            object3 == "a fresh cookie, still warm and all.")
    {
        result.innerHTML =
        "The demon appears! It seems way more interested in the cookie than helping with your homework.";
    }

    else{
        result.innerHTML =
        "Nothing happens. Now your room smells weird and you still have homework to do.";
    }
}

function resetGame(){

    document.getElementById("object1").selectedIndex = 0;
    document.getElementById("object2").selectedIndex = 0;
    document.getElementById("object3").selectedIndex = 0;

    document.getElementById("result").innerHTML = "";
}
